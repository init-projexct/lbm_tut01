cmake_minimum_required(VERSION 3.10)
project(lbm_tut01)

# Stick to C++17. It is more sustainable and avoids the 'std::views' bug in Palabos.
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(PALABOS_ROOT /home/researchx/apps/palabos)

include_directories(SYSTEM ${PALABOS_ROOT}/src)
include_directories(SYSTEM ${PALABOS_ROOT}/externalLibraries)

# 1. COLLECT ALL SOURCES
file(GLOB_RECURSE PALABOS_SOURCES 
    "${PALABOS_ROOT}/src/*.cpp"
    "${PALABOS_ROOT}/externalLibraries/*.cpp"
)

# 2. CREATE A LIBRARY (This speeds up building multiple executables)
add_library(palabos_lib STATIC ${PALABOS_SOURCES})

# 3. COMPILER OPTIONS & DEFINITIONS
# PLB_USE_CPP20=OFF is the crucial fix for the std::views error.
target_compile_definitions(palabos_lib PRIVATE PLB_USE_CPP20=OFF)
target_compile_options(palabos_lib PRIVATE 
    -O3 
    -march=native
    -fpermissive
    -Wno-narrowing
    -Wno-ignored-attributes
)

# 4. CREATE EXECUTABLES
add_executable(main main.cpp)
add_executable(test01 test01.cpp)

# 5. LINKING
find_package(MPI REQUIRED)
set(LIBRARIES_TO_LINK palabos_lib MPI::MPI_CXX pthread tinyxml)

target_link_libraries(main PRIVATE ${LIBRARIES_TO_LINK})
target_link_libraries(test01 PRIVATE ${LIBRARIES_TO_LINK})
